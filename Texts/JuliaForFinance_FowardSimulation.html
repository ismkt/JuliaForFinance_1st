using Distributions
using StatsPlots

function CalcForwardValue(S, df, K)
    return S - df * K
end

#Tt
#r
function CalcDiscountFactor(r, T, t)
    return exp(-r * (T - t))
end

function SimulateForwardValue()
    #1.
    T = 1
    t = 0.01
     = 0.5
     = 0.2
    S0 = 100.0 #
    K = 120
    r = 0.01 #

    #2.
    t = 0:t:T

    #3.
    d = Normal(0, t^0.5)

    #4.
    S = [S0]
    df = CalcDiscountFactor(r, T, t[1])
    V0 = CalcForwardValue(S[1], df, K)
    V = [V0]
    
    #5.
    for t_i in t[2:end]
        constantDiff =  * t * S[end] #
        w = rand(d) #
        randomDiff =  * w * S[end] #
        nextS = S[end] + constantDiff  + randomDiff #
        push!(S, nextS) #

        #6.
        df = CalcDiscountFactor(r, T, t_i)

        #7.
        nextV = CalcForwardValue(nextS, df, K)
        push!(V, nextV)
    end

    #8.
    p1 = plot(t, S, label = "StockPrice") #
    plot!(p1, t, fill(K, length(t)), label = "K") #K. fill(K, n)nK

    p2 = plot(t, V, label = "ForwardValue")#

    plot(p1, p2)
end
