using Distributions
using StatsPlots

function SimulateStockPrice()
    #1.
    T = 1
    t = 0.01
     = 0.5
     = 0.2
    S0 = 100.0#

    #2.
    t = 0:t:T

    #3.
    d = Normal(0, t^0.5)

    #4.
    S = [S0]

    #4.
    for t_i in t[2:end]
        constantDiff =  * t * S[end] #
        w = rand(d) #
        randomDiff =  * w * S[end] #
        nextS = S[end] + constantDiff  + randomDiff #
        push!(S, nextS) #
    end

    #5.
    plot(t, S, label = "StockPrice")
end

using Distributions
using StatsPlots

#n
function SimulateMultiStockPrice(n)
    #1.
    T = 1
    t = 0.01
     = 0.5
     = 0.2
    S0 = 100.0#

    #2.
    t = 0:t:T

    #3.
    d = Normal(0, t^0.5)

    p = plot() #

    for i in 1:n
        #4.
        S = [S0]
        for t_i in t[2:end]
            constantDiff =  * t * S[end] #
            w = rand(d) #
            randomDiff =  * w * S[end] #
            nextS = S[end] + constantDiff  + randomDiff #
            push!(S, nextS) #
        end

        #5.
        plot!(p, t, S, label = "StockPrice_" * string(i))
    end

    plot(p)
end
